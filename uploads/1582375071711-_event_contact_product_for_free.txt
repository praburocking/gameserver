public DataObject getTabsDO(Set<Long> tabIds)throws Exception
{
	SelectQuery sq = new SelectQueryImpl(Table.getTable(CRMTAB.TABLE));
	sq.addSelectColumn(Column.getColumn(null,"*"));
	sq.setCriteria(new Criteria(Column.getColumn(CRMTAB.TABLE, CRMTAB.TABID), tabIds.toArray(), QueryConstants.IN));
	return PersistenceCacheUtil.getPurePersistenceLiteHandle().get(sq);
}
